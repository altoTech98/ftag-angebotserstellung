class MatchDimension(str, Enum):
    MASSE = "Masse"
    BRANDSCHUTZ = "Brandschutz"
    SCHALLSCHUTZ = "Schallschutz"
    MATERIAL = "Material"
    ZERTIFIZIERUNG = "Zertifizierung"
    LEISTUNG = "Leistung"

class DimensionScore(BaseModel):
    dimension: MatchDimension
    score: float  # 0.0 - 1.0
    begruendung: str

class MatchCandidate(BaseModel):
    produkt_id: str
    produkt_name: str
    gesamt_konfidenz: float  # 0.0 - 1.0
    dimension_scores: list[DimensionScore]
    begruendung: str

class MatchResult(BaseModel):
    positions_nr: str
    bester_match: Optional[MatchCandidate]
    alternative_matches: list[MatchCandidate]
    hat_match: bool  # True only when gesamt_konfidenz >= 0.95
    match_methode: str  # 'tfidf_ai', 'keyword_fallback', 'manual'
